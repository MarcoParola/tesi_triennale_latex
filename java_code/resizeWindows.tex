     private void resize_windows(Long time) {

            // ACCELEROMETR
            while(accelerometr_signal_timestamp.get(1) < time) {
                  accelerometr_signal_timestamp.remove(0);
                  accelerometr_signal.remove(0);

            if((time - accelerometr_signal_timestamp.get(0)) >
